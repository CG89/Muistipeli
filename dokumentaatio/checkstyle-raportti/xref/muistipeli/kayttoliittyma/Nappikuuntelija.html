3   import java.awt.event.ActionEvent;
4   import java.awt.event.ActionListener;
5   import javax.swing.ImageIcon;
6   import javax.swing.JButton;
7   import javax.swing.JFrame;
8   import muistipeli.ohjelmalogiikka.Lauta;
9   
10  public class Nappikuuntelija implements ActionListener {
11  
12      private JButton nappi;
13      private int indeksi;
14      private Lauta lauta;
15      private JFrame frame;
16  
17      public Nappikuuntelija(JButton nappi, JFrame frame, Lauta lauta) {
18          this.indeksi = Integer.parseInt(nappi.getText()) - 1;
19          this.lauta = lauta;
20          this.nappi = nappi;
21          this.frame = frame;
22  
23      }
24  
25      @Override
26      public void actionPerformed(ActionEvent e) {
27  
28          lauta.suljeVanhatAvatutKortit();
29          int vuoronNumero = lauta.getVuoronNumero();
30          String palautus = lauta.korttiKaannetty(indeksi, nappi);
31          frame.setTitle("Vuoro #" + vuoronNumero + ": " + lauta.getTietokanta().getAktiivinenPelaaja() + ", " + lauta.getTietokanta().getAktiivinenPelaaja().getPistetilanne() + " löydetty ");
32          nappi.setText("");
33          nappi.setIcon(new ImageIcon(lauta.getTietokanta().getSekoitetutKortit().get(indeksi).getKuva()));
34          if (palautus.equals("Voitit pelin, onneksi olkoon!")) {
35              if (lauta.getPelaajamaara() == 2) {
36                  frame.setTitle("Lopulliset pisteet: " + lauta.getTietokanta().getPelaajat().get(0) + ": " + lauta.getTietokanta().getPelaajat().get(0).getPistetilanne() + " | "
37                          + lauta.getTietokanta().getPelaajat().get(1) + ": " + lauta.getTietokanta().getPelaajat().get(1).getPistetilanne());
38              } else {
39                  frame.setTitle("Löysit kaikki parit " + lauta.getVuoronNumero() + " vuorossa!");
40  
41              }
42          }
43      }
44  
45  }
