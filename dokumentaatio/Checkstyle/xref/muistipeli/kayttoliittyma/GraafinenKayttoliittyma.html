35      public void run() {
36          menunLuonti();
37          frame = new JFrame("Muistipeli");

46      public void menunLuonti() {
47          menuBar = new JMenuBar();
48          menu = new JMenu("Valikko");
49          menu.setMnemonic(KeyEvent.VK_A);
50          menuBar.add(menu);
51          menuItem = new JMenuItem("Uusi peli",
52                  KeyEvent.VK_T);
53          menuItem.setAccelerator(KeyStroke.getKeyStroke(
54                  KeyEvent.VK_1, ActionEvent.ALT_MASK));
55          menuItem.addActionListener(new Menukuuntelija(menuItem,this));
56          menu.add(menuItem);
57          submenu = new JMenu("Vaikeusaste");
58          submenu.setMnemonic(KeyEvent.VK_S);
59          menuItem = new JMenuItem("Helppo");
60          menuItem.setAccelerator(KeyStroke.getKeyStroke(
61                  KeyEvent.VK_3, ActionEvent.ALT_MASK));
62          menuItem.addActionListener(new Menukuuntelija(menuItem,this));
63          submenu.add(menuItem);
64          menuItem = new JMenuItem("Keskitaso");
65          menuItem.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_4, ActionEvent.ALT_MASK));
66          submenu.add(menuItem);
67          menuItem.addActionListener(new Menukuuntelija(menuItem,this));
68          menu.add(submenu);
69          menuItem = new JMenuItem("Vaikea");
70          menuItem.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_5,ActionEvent.ALT_MASK));
71          submenu.add(menuItem);
72          menuItem.addActionListener(new Menukuuntelija(menuItem,this));
73          menu.add(submenu);
74      }

76      public void luoKomponentit(Container container) {
77          GridLayout layout = new GridLayout(lauta.getVaikeusaste(), lauta.getVaikeusaste(), 0, 0);
78          container.setLayout(layout);
79          for (int i = 0; i < lauta.getVaikeusaste()*4; i++) {
80              String j = String.valueOf(i);
81              JButton lisattava = new JButton(j);
82              lisattava.setFont(new Font("Arial", Font.PLAIN, 80));
83              lisattava.addActionListener(new Nappikuuntelija(lisattava, frame, lauta));
84              container.add(lisattava, i);
85          }
86      }
